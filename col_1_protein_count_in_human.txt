ip_distribution = ip_distribution_0718


for i = 1,1

  t1 = all_human_ip.ip_code ==ip_distribution.Domain_ID[i]

  group t1.uniprot t1.uniprot 'count,count' name='temp'
  delete t1
  delete temp.count
  add column temp Sarray(Nof(temp),"") index=2 name="Gene Name Primary" append 
  add column temp Sarray(Nof(temp),"") index=3 name="Protein Name" append  
  add column temp Sarray(Nof(temp),"") index=4 name="Gene Name Synonym" append 
  add column temp Count(1 Nof(temp)) index=1 name="index" append 
  rename temp.uniprot "Uniprot_ID"

  for j = 1, Nof(temp)

    t2 = temp.Uniprot_ID[j] ==uniprot_full_info.Entry

    if Nof(t2)!=0 then
      temp.Gene_Name_Primary[j] = t2.Gene_names_primary_[1]
      temp.Protein_Name[j]=t2.Protein_names[1]
      temp.Gene_Name_Synonym[j]=t2.Gene_names_synonym_[1]
    endif

  endfor
  f_name = "col_1_"+ip_distribution.Domain_ID[i]
  write html temp f_name format = yes header = yes frame = 1 chemical = yes sort view size = 200 // 150
  rename temp f_name

  
endfor
delete t2, ip_distribution
